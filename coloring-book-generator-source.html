üß† AI Content Generation: Two-Stage Creative Pipeline

The AI generation system uses a sophisticated two-stage process: intelligent theme interpretation followed by visual creation, ensuring thematic consistency across all coloring pages:

ü§ñ geminiService.ts
import { GoogleGenAI, Type } from "@google/genai"; const API_KEY = process.env.API_KEY; if (!API_KEY) { throw new Error("API_KEY environment variable not set"); } const ai = new GoogleGenAI({ apiKey: API_KEY }); // Stage 1: Intelligent Theme Interpretation & Idea Generation export async function generatePageIdeas(theme: string): Promise { try { // Advanced prompt engineering for creative consistency const response = await ai.models.generateContent({ model: "gemini-2.5-flash", contents: `Generate a list of 10 distinct, child-friendly subjects for coloring pages with the theme "${theme}". The subjects should be single, simple concepts. For example, for "sea animals", you could return ["A friendly dolphin", "A curious sea turtle", "A starfish on the sand"].`, config: { responseMimeType: "application/json", responseSchema: { type: Type.OBJECT, properties: { ideas: { type: Type.ARRAY, items: { type: Type.STRING, description: 'A single coloring page subject or idea.', }, }, }, required: ['ideas'], }, }, }); // Structured JSON response validation const jsonText = response.text.trim(); const result = JSON.parse(jsonText); if (result.ideas && Array.isArray(result.ideas) && result.ideas.length >= 10) { return result.ideas.slice(0, 10); } else { throw new Error("Invalid or insufficient ideas generated."); } } catch (error) { console.error("Error generating page ideas:", error); throw new Error("Failed to get ideas from AI. Please check the theme and try again."); } } // Stage 2: Visual Generation with Age-Appropriate Optimization export async function generateImage( subject: string, agePrompt: string, stylePrompt: string ): Promise { // Advanced prompt engineering for print-optimized coloring pages const prompt = `A black and white coloring book page for a child. The style is ${stylePrompt} and ${agePrompt}. The drawing is of: "${subject}". It must have clean lines, no shading or color, on a pure white background. The image should be centered and fill the space well.`; try { // Imagen 4.0 for high-quality line art generation const response = await ai.models.generateImages({ model: 'imagen-4.0-generate-001', prompt: prompt, config: { numberOfImages: 1, outputMimeType: 'image/png', aspectRatio: '1:1', // Perfect for coloring books }, }); if (response.generatedImages && response.generatedImages.length > 0) { return response.generatedImages[0].image.imageBytes; } else { throw new Error("No image was generated."); } } catch (error) { console.error(`Error generating image for subject "${subject}":`, error); // Comprehensive error handling for production reliability throw new Error(`Failed to generate an image for "${subject}".`); } }

// Professional PDF generation with print optimization export async function generatePdf(pages: Page[], theme: string): Promise { try { // Dynamic import for optimal bundle size const { jsPDF } = await import('jspdf'); const html2canvas = (await import('html2canvas')).default; // A4 format configuration for optimal printing const pdf = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' }); for (let i = 0; i < pages.length; i++) { const page = pages[i]; const element = document.getElementById(page.id); if (element) { // High-resolution canvas capture for crisp printing const canvas = await html2canvas(element, { scale: 2, // High DPI for print quality useCORS: true, backgroundColor: '#ffffff', width: 794, // A4 width in pixels at 96 DPI height: 1123 // A4 height in pixels at 96 DPI }); const imageData = canvas.toDataURL('image/png'); // Add new page for each coloring page (except first) if (i > 0) { pdf.addPage(); } // Add image with proper margins and scaling pdf.addImage(imageData, 'PNG', 10, 10, 190, 277); } } // Professional filename with theme and timestamp const filename = `${theme.replace(/[^a-z0-9]/gi, '_').toLowerCase()}_coloring_book.pdf`; pdf.save(filename); } catch (error) { console.error('PDF Generation Error:', error); throw new Error('Failed to generate PDF. Please try again.'); } }
