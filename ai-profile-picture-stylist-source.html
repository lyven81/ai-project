import { GoogleGenerativeAI } from '@google/genai'; import { STYLE_PROMPTS } from '../constants'; import { GeneratedImage } from '../types'; // Initialize Google Gemini AI with secure API key const genAI = new GoogleGenerativeAI(import.meta.env.VITE_GEMINI_API_KEY); /** * Advanced people detection using Google Gemini's vision capabilities * Analyzes image content to count people with high accuracy */ export const countPeopleInImage = async (imageDataUrl: string): Promise<number> => { const model = genAI.getGenerativeModel({ model: 'gemini-1.5-flash' }); // Convert base64 image for Gemini processing const imagePart = { inlineData: { data: imageDataUrl.split(',')[1], // Remove data:image/jpeg;base64, prefix mimeType: 'image/jpeg' } }; // Specialized prompt for accurate people counting const countPrompt = ` Analyze this image carefully and count the number of people visible. Instructions: - Count only clearly visible human faces and bodies - Include people who are partially visible but recognizable - Do not count reflections, shadows, or unclear figures - Respond with only a single number (0, 1, 2, 3, etc.) Image analysis focus: - Look for faces, heads, full or partial human bodies - Consider different poses, angles, and lighting conditions - Be precise and conservative in counting `; try { const result = await model.generateContent([countPrompt, imagePart]); const response = await result.response; const text = response.text().trim(); // Parse and validate the count response const count = parseInt(text); return isNaN(count) ? 0 : count; } catch (error) { console.error('Error counting people:', error); throw new Error('Failed to analyze image content'); } }; /** * Professional image style generation system * Creates 4 distinct professional variations using specialized AI prompts */ export const generateStyledImages = async ( imageDataUrl: string, personCount: number ): Promise<GeneratedImage[]> => { const model = genAI.getGenerativeModel({ model: 'gemini-1.5-flash' }); // Prepare image for AI processing const imagePart = { inlineData: { data: imageDataUrl.split(',')[1], mimeType: 'image/jpeg' } }; // Define professional style configurations const styles = [ { name: 'Classic Black & White', description: 'Timeless monochrome professional styling', prompt: STYLE_PROMPTS.CLASSIC_BW }, { name: 'Corporate Executive', description: 'Professional business headshot appearance', prompt: STYLE_PROMPTS.CORPORATE }, { name: 'Natural Lifestyle', description: 'Warm, approachable lifestyle photography', prompt: STYLE_PROMPTS.LIFESTYLE }, { name: 'Modern Editorial', description: 'Contemporary artistic portrait styling', prompt: STYLE_PROMPTS.EDITORIAL } ]; // Generate all styles concurrently for optimal performance const stylePromises = styles.map(async (style, index) => { try { // Construct specialized prompt based on person count and style const contextualPrompt = ` Transform this image into a professional ${style.name.toLowerCase()} style.
